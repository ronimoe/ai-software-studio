# v0.1 Plan 7 — GitHub push + `gh pr create` + PR evidence report

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Close the v0.1 loop. From the Review Room, the user can (1) "Copy PR report" → a Markdown evidence report goes to the clipboard, and (2) "Create PR" → the worktree branch is pushed via `git push -u origin <branch>` and a PR is created via `gh pr create --title --body --base`, returning a URL the user can click.

**Architecture:** `GitHubService` shells `gh` for detection (`which gh`, `gh auth status`), branch push, and PR creation. A pure renderer in `commands::pr::report.rs` composes a Markdown report from `Task`, `ChangedFile[]`, latest `VerificationRun`. Three Tauri commands: `detect_github`, `render_pr_report`, `create_pr`. Frontend: two new buttons in the Review Room actions cluster, each with appropriate gating.

**Tech Stack:** `gh` CLI (subprocess), Tauri clipboard plugin for the copy action, no new Rust crates.

use crate::{error::AppError, models::{GitHubAuthStatus, GitHubStatus}};
use std::path::Path;
use std::process::Command;

pub fn detect() -> Result<GitHubStatus, AppError> {
    let Some(bin) = which("gh") else {
        return Ok(GitHubStatus {
            auth: GitHubAuthStatus::NotInstalled,
            binary_path: None,
            account: None,
        });
    };
    let auth_out = Command::new(&bin)
        .args(["auth", "status"])
        .output()
        .map_err(|e| AppError::internal(format!("gh auth status: {e}")))?;

    let combined = format!(
        "{}\n{}",
        String::from_utf8_lossy(&auth_out.stdout),
        String::from_utf8_lossy(&auth_out.stderr)
    );
    let account = parse_account(&combined);

    let auth = if auth_out.status.success() {
        GitHubAuthStatus::Authed
    } else {
        GitHubAuthStatus::NotAuthed
    };

    Ok(GitHubStatus {
        auth,
        binary_path: Some(bin),
        account,
    })
}

pub fn push_branch(repo: &Path, branch: &str) -> Result<(), AppError> {
    let out = Command::new("git")
        .args(["push", "-u", "origin", branch])
        .current_dir(repo)
        .output()
        .map_err(|e| AppError::internal(format!("git push: {e}")))?;
    if !out.status.success() {
        return Err(AppError::internal(format!(
            "git push -u origin {branch} failed: {}",
            String::from_utf8_lossy(&out.stderr).trim()
        )));
    }
    Ok(())
}

pub fn create_pr(
    repo: &Path,
    title: &str,
    body: &str,
    base: &str,
    draft: bool,
) -> Result<String, AppError> {
    let mut args: Vec<&str> = vec!["pr", "create", "--title", title, "--body", body, "--base", base];
    if draft {
        args.push("--draft");
    }
    let out = Command::new("gh")
        .args(&args)
        .current_dir(repo)
        .output()
        .map_err(|e| AppError::internal(format!("gh pr create: {e}")))?;
    if !out.status.success() {
        return Err(AppError::internal(format!(
            "gh pr create failed: {}",
            String::from_utf8_lossy(&out.stderr).trim()
        )));
    }
    let stdout = String::from_utf8_lossy(&out.stdout);
    let url = stdout
        .lines()
        .find(|l| l.contains("github.com"))
        .map(|l| l.trim().to_string())
        .ok_or_else(|| AppError::internal(format!("gh pr create returned no URL: {}", stdout.trim())))?;
    Ok(url)
}

fn which(name: &str) -> Option<String> {
    let path = std::env::var_os("PATH")?;
    for dir in std::env::split_paths(&path) {
        let candidate = dir.join(name);
        if candidate.is_file() {
            #[cfg(unix)]
            {
                use std::os::unix::fs::PermissionsExt;
                if let Ok(meta) = std::fs::metadata(&candidate) {
                    if meta.permissions().mode() & 0o111 != 0 {
                        return Some(candidate.to_string_lossy().into_owned());
                    }
                }
            }
            #[cfg(not(unix))]
            { return Some(candidate.to_string_lossy().into_owned()); }
        }
    }
    None
}

fn parse_account(s: &str) -> Option<String> {
    // Match `Logged in to github.com as someuser` (with or without parens trailing).
    for line in s.lines() {
        if let Some(idx) = line.find(" as ") {
            let after = &line[idx + 4..];
            let acct: String = after
                .chars()
                .take_while(|c| !c.is_whitespace() && *c != '(' && *c != ',')
                .collect();
            if !acct.is_empty() {
                return Some(acct);
            }
        }
    }
    None
}

#[tauri::command]
#[specta::specta]
pub async fn render_pr_report(
    state: State<'_, AppState>,
    task_id: String,
) -> Result<String, AppError> {
    let task = state.tasks.get(&task_id).await?;
    let runs = state.verification.list_for_task(&task_id).await?;
    let latest = runs.first().cloned();

    // changed files
    let worktree = task.worktree_path.clone();
    let changed = if let Some(wt) = worktree.clone() {
        let p = PathBuf::from(wt);
        tokio::task::spawn_blocking(move || crate::git::status::status(&p))
            .await
            .map_err(|e| AppError::internal(format!("join: {e}")))??
    } else {
        Vec::new()
    };

    Ok(report::render(&task, &changed, latest.as_ref()))
}

#[tauri::command]
#[specta::specta]
pub async fn create_pr(
    state: State<'_, AppState>,
    request: CreatePrRequest,
) -> Result<PrResult, AppError> {
    let task = state.tasks.get(&request.task_id).await?;
    let project = state.projects.get(&task.project_id).await?;
    let branch = task
        .branch_name
        .clone()
        .ok_or_else(|| AppError::invalid_arg("task has no branch; create a worktree first"))?;
    let worktree = task
        .worktree_path
        .clone()
        .ok_or_else(|| AppError::invalid_arg("task has no worktree"))?;
    let base = request
        .base_branch
        .clone()
        .unwrap_or(project.default_branch.clone());

    // Push from the worktree (not the main checkout).
    let repo_path = PathBuf::from(&worktree);
    let branch_clone = branch.clone();
    tokio::task::spawn_blocking(move || crate::engines::github::push_branch(&repo_path, &branch_clone))
        .await
        .map_err(|e| AppError::internal(format!("join: {e}")))??;

    // Render body
    let runs = state.verification.list_for_task(&task.id).await?;
    let latest = runs.first().cloned();
    let changed = {
        let p = PathBuf::from(&worktree);
        tokio::task::spawn_blocking(move || crate::git::status::status(&p))
            .await
            .map_err(|e| AppError::internal(format!("join: {e}")))??
    };
    let body = report::render(&task, &changed, latest.as_ref());

    // Create PR (from the worktree)
    let repo_path = PathBuf::from(&worktree);
    let title = task.title.clone();
    let base_clone = base.clone();
    let url = tokio::task::spawn_blocking(move || {
        crate::engines::github::create_pr(&repo_path, &title, &body, &base_clone, request.draft)
    })
    .await
    .map_err(|e| AppError::internal(format!("join: {e}")))??;

    state.tasks.update_status(&task.id, crate::models::TaskStatus::PrPrepared).await?;

    Ok(PrResult { url, branch, base })
}

use crate::models::{ChangedFile, Task, VerificationRun, VerificationStatus};
use std::fmt::Write as _;

pub fn render(task: &Task, changed: &[ChangedFile], latest_run: Option<&VerificationRun>) -> String {
    let mut out = String::new();
    writeln!(out, "# AI Software Studio Evidence Report").unwrap();
    writeln!(out).unwrap();
    writeln!(out, "## Task").unwrap();
    writeln!(out).unwrap();
    writeln!(out, "**{}** (`{}`)", task.title, task.id).unwrap();
    writeln!(out).unwrap();

    if !task.description.trim().is_empty() {
        writeln!(out, "{}", task.description.trim()).unwrap();
        writeln!(out).unwrap();
    }

    writeln!(out, "## Acceptance Criteria").unwrap();
    writeln!(out).unwrap();
    if task.acceptance_criteria.is_empty() {
        writeln!(out, "_None defined._").unwrap();
    } else {
        for ac in &task.acceptance_criteria {
            let mark = if ac.satisfied { "x" } else { " " };
            writeln!(out, "- [{mark}] {}", ac.label).unwrap();
        }
    }
    writeln!(out).unwrap();

    writeln!(out, "## Files Changed").unwrap();
    writeln!(out).unwrap();
    if changed.is_empty() {
        writeln!(out, "_No changed files detected._").unwrap();
    } else {
        writeln!(out, "| File | Status |").unwrap();
        writeln!(out, "|---|---|").unwrap();
        for c in changed {
            writeln!(out, "| `{}` | {:?} |", c.path, c.status).unwrap();
        }
    }
    writeln!(out).unwrap();

    writeln!(out, "## Verification").unwrap();
    writeln!(out).unwrap();
    match latest_run {
        None => writeln!(out, "_No verification run yet._").unwrap(),
        Some(run) => {
            writeln!(out, "Run `{}` started at {}.", run.id, run.started_at).unwrap();
            writeln!(out).unwrap();
            writeln!(out, "| Check | Status | Duration |").unwrap();
            writeln!(out, "|---|---|---|").unwrap();
            for c in &run.checks {
                let badge = match c.status {
                    VerificationStatus::Passed => "✅ passed",
                    VerificationStatus::Failed => "❌ failed",
                    VerificationStatus::Skipped => "⏭ skipped",
                    VerificationStatus::Warning => "⚠️ warning",
                    VerificationStatus::Running => "⏳ running",
                    VerificationStatus::NotRun => "— not run",
                };
                let dur = c.duration_ms.map(|m| format!("{m}ms")).unwrap_or_else(|| "—".to_string());
                writeln!(out, "| {} | {} | {} |", c.kind, badge, dur).unwrap();
            }
        }
    }
    writeln!(out).unwrap();

    if !task.constraints.is_empty() {
        writeln!(out, "## Constraints").unwrap();
        writeln!(out).unwrap();
        for c in &task.constraints {
            writeln!(out, "- {c}").unwrap();
        }
        writeln!(out).unwrap();
    }

    writeln!(out, "---").unwrap();
    writeln!(out, "Generated by AI Software Studio · task `{}`", task.id).unwrap();

    out
}

export function useCreatePr() {

"use client";

import { useState } from "react";
import { Loader2, GitPullRequest, ExternalLink } from "lucide-react";
import { Button } from "@/components/ui/button";
import { useDetectGithub } from "@/features/pr/use-detect-github";
import { useCreatePr } from "@/features/pr/use-create-pr";

interface Props {
  taskId: string;
  hasWorktree: boolean;
}

export function CreatePrButton({ taskId, hasWorktree }: Props) {
  const { data: gh } = useDetectGithub();
  const create = useCreatePr();
  const [prUrl, setPrUrl] = useState<string | null>(null);

  const disabled =
    !hasWorktree ||
    gh?.auth !== "authed" ||
    create.isPending;

  const tooltip =
    !hasWorktree
      ? "Create a worktree first"
      : gh?.auth === "notInstalled"
        ? "`gh` not installed — `brew install gh`"
        : gh?.auth === "notAuthed"
          ? "Run `gh auth login` first"
          : "Push branch and open a PR";

  const onClick = async () => {
    const pr = await create.mutateAsync({ taskId, baseBranch: null, draft: false });
    setPrUrl(pr.url);
  };

  if (prUrl) {
    return (
      <a href={prUrl} target="_blank" rel="noreferrer">
        <Button size="sm" variant="default">
          <ExternalLink className="mr-1 h-3 w-3" />
          Open PR
        </Button>
      </a>
    );
  }

  return (
    <Button size="sm" onClick={onClick} disabled={disabled} title={tooltip}>
      {create.isPending ? (
        <Loader2 className="mr-1 h-3 w-3 animate-spin" />
      ) : (
        <GitPullRequest className="mr-1 h-3 w-3" />
      )}
      Create PR
    </Button>
  );
}
